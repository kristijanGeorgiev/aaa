import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      title: 'Pull to Refresh Colors',
      debugShowCheckedModeBanner: false,
      home: ColorListScreen(),
    );
  }
}

class ColorListScreen extends StatefulWidget {
  const ColorListScreen({super.key});

  @override
  State<ColorListScreen> createState() => _ColorListScreenState();
}

class _ColorListScreenState extends State<ColorListScreen> {
  List<String> _colors = ["Red", "Green", "Blue"];
  bool _hasRefreshed = false;

  Future<void> _handleRefresh() async {
    await Future.delayed(const Duration(seconds: 1)); // Simulate delay

    if (!_hasRefreshed) {
      setState(() {
        _colors = ["Cyan", "Magenta", "Yellow"];
        _hasRefreshed = true;
      });
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text("New colors loaded")),
      );
    } else {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text("No more updates")),
      );
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("Color List")),
      body: RefreshIndicator(
        onRefresh: _handleRefresh,
        child: ListView.builder(
          itemCount: _colors.length,
          itemBuilder: (context, index) {
            return ListTile(
              title: Text(_colors[index]),
            );
          },
        ),
      ),
    );
  }
}
